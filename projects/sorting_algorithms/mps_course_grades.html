# Create hierarchical index for labs and tasks
lab_labels = []
task_labels = []
# How many tasks in each lab
lab_tasks = {1: 4, 2: 3, 3: 3, 4: 3, 5: 3}
for lab in lab_tasks:
    for task in range(lab_tasks[lab]):
        lab_labels.append(f'L{lab}')
        task_labels.append(f'T{task+1}')

ind_tuples = list(zip(*[lab_labels, task_labels]))
index = pd.MultiIndex.from_tuples(ind_tuples, names=['Lab', 'Task'])
grades.columns = index

def group_students(lab):
    """Gather students who scored exactly the same on given lab into groups."""
    group_grades = grades.assign(temp=grades[lab].sum(axis=1)) \
        .sort_values('temp', ascending=False)[lab]
    changes = (group_grades.pct_change()).any(axis=1)
    group_grades['Group'] = ''
    group_tag = 65
    for ind in range(0, len(group_grades)):
        name = group_grades.index[ind]
        group_grades.loc[name, 'Group'] = chr(group_tag)
        if ind < len(group_grades) - 1:
            if changes[ind + 1]:
                group_tag += 1
    # group_grades.rename_axis(columns = 'Tasks')
    return group_grades

def group_students_all():
    """Gather students who scored exactly the same on all labs into groups."""
    group_grades = grades.assign(temp=grades.sum(axis=1)) \
        .sort_values('temp', ascending=False).drop('temp', 1)
    changes = (group_grades.pct_change()).any(axis=1)
    group_grades['Group'] = ''
    group_tag = 65
    for ind in range(0, len(group_grades)):
        name = group_grades.index[ind]
        group_grades.loc[name, 'Group'] = chr(group_tag)
        if ind < len(group_grades) - 1:
            if changes[ind + 1]:
                group_tag += 1
            if group_tag == 90:
                group_tag = 97
    return group_grades

    if group_code in groups.keys():
        groups[group_code].append(name)
    else:
        groups[group_code] = [name]
